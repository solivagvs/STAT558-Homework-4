getNews = function(querystr = NA, dateStr = Sys.Date(), apiKey = "44c616340dfb495f90ed954af0159a95"){
  if (!is.na(querystr)){
    query = paste(querystr,"&", sep = "")
  }
  else{
    query = ""
  }
  base_url = "https://newsapi.org/v2/everything?q="
  
  dateURL = paste0("from=",dateStr, sep = "")
  middleURL = "&language=en&pageSize=50&"
  apiURL = paste("apiKey=", apiKey, sep = "")
  
  url = paste(base_url, query,dateURL,middleURL, apiURL, sep = "")
  
  news = fromJSON(rawToChar(GET(url)$content))$articles
  
  return (news)
}
